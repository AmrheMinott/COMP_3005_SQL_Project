
create table users(
	u_id		varchar(21),
	bill_info	varchar(21),
	ship_info	varchar(21),

	primary key (u_id)
);


create table publishers(
  name		varchar(21),
	address		varchar(55),
  phone_num	varchar(13),
  bank_acc	numeric(8),
  pub_id		varchar(8),

	primary key (pub_id)

);


create table books(
  author		varchar(21),
	genre		varchar(13),
  pub_id		varchar(8),
  num_of_pages	numeric(5,0) check (num_of_pages > 0),
  price		numeric(5,0) check (price > 0),
  isbn		varchar(8),
  title		varchar(21),
	percent		numeric(5,0) check (percent > 0),
	pub_name	varchar(21),
	quantity	numeric(5,0),

	primary key (isbn),
	foreign key (pub_id) references publishers (pub_id)
		on delete cascade

);


create table bookstore(
  bill_info	varchar(21),
	ship_info	varchar(21),
  order_num	varchar(21),
  isbn		varchar(8),
	u_id		varchar(21),

	primary key (order_num),
	foreign key (bill_info) references users
		on delete cascade,
	foreign key (ship_info) references users
		on delete cascade,
	foreign key (u_id) references users
		on delete cascade,

	foreign key (isbn) references books (isbn)
		on delete cascade

);


create table warehouse(
	bill_info	varchar(21),
	ship_info	varchar(21),
	order_num	varchar(21),
	tracking_info	serial primary key,

	foreign key (bill_info) references bookstore
		on delete cascade,
	foreign key (ship_info) references bookstore
		on delete cascade,
	foreign key (order_num) references bookstore
		on delete cascade

);
